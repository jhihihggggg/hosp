┌──────────────────────────────────────────────────────────────────────────────┐
│  ⚡ QUICK COMMANDS                                                           │
└──────────────────────────────────────────────────────────────────────────────┘

  Start/Stop Service
  ──────────────────
  systemctl start hospital
  systemctl stop hospital
  systemctl restart hospital
  systemctl status hospital

  View Logs
  ──────────────────
  journalctl -u hospital -f

  Django Commands (on VPS)
  ──────────────────
  cd /var/www/hospital
  source venv/bin/activate
  python manage.py [command] --settings=diagcenter.production_settings

  Nginx
  ──────────────────
  nginx -t
  systemctl reload nginx


┌──────────────────────────────────────────────────────────────────────────────┐
│  🔧 COMMON TASKS                                                             │
└──────────────────────────────────────────────────────────────────────────────┘

  Create Superuser
  ──────────────────
  cd /var/www/hospital && source venv/bin/activate
  python manage.py createsuperuser --settings=diagcenter.production_settings

  Backup Database
  ──────────────────
  cp /var/www/hospital/hospital_production.db \
     /root/backups/hospital_$(date +%Y%m%d).db

  Update Code
  ──────────────────
  1. ./upload_to_vps.sh
  2. systemctl restart hospital


┌──────────────────────────────────────────────────────────────────────────────┐
│  🐛 TROUBLESHOOTING                                                          │
└──────────────────────────────────────────────────────────────────────────────┘

  Service Won't Start
  ──────────────────
  journalctl -u hospital -xe
  chown -R www-data:www-data /var/www/hospital
  systemctl restart hospital

  502 Bad Gateway
  ──────────────────
  systemctl status hospital
  systemctl restart hospital
  systemctl reload nginx

  Check Port
  ──────────────────
  netstat -tulpn | grep 8002

  Fix Permissions
  ──────────────────
  chown -R www-data:www-data /var/www/hospital
  chmod -R 755 /var/www/hospital
